241def PreStepUF(mbs, t):
242    if compensateStaticTorques:
243        staticTorques = ComputeMBSstaticRobotTorques(robot)
244        #print("tau=", staticTorques)
245    else:
246        staticTorques = np.zeros(len(jointList))

254            phi = mbs.GetObjectOutput(joint, exu.OutputVariableType.Rotation)[2] #z-rotation
255            omega = mbs.GetObjectOutput(joint, exu.OutputVariableType.AngularVelocityLocal)[2] #z-angular velocity
256            tsd = torsionalSDlist[i]
257            mbs.SetObjectParameter(tsd, 'offset', u[i])
258            mbs.SetObjectParameter(tsd, 'velocityOffset', v[i])
259            mbs.SetObjectParameter(tsd, 'torque', staticTorques[i]) #additional torque from given velocity
260    else:
261        #in case of kinematic tree, very simple operations!
262        mbs.SetObjectParameter(oKT, 'jointPositionOffsetVector', u)
263        mbs.SetObjectParameter(oKT, 'jointVelocityOffsetVector', v)
264        mbs.SetObjectParameter(oKT, 'jointForceVector', staticTorques)
265
266    return True
