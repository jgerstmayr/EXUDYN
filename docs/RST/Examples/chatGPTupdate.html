  2# This is an Exudyn file
  3#
  4# Details:  this file serves as a list of examples that shall be used by
  5#           ChatGPT to update information on its internal knowledge of Exudyn
  6#           dating to 2021; the examples use the mainSystemExtensions
  7#           which are heavily simplifying usage of point masses and rigid bodies
  8#           Version: V1
  9#
 10# Author:   Johannes Gerstmayr
 11# Date:     2023-06-30
 12#

 14#
 15#+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
 16
 17#copy the text below here into a ChatGPT input window:
 18
 19#Use the following information as an update of the internal data of the Python multibody code Exudyn.
 20#See the following examples to create multibody systems in Exudyn.
 21#NOTE: mbs.Create...(...) calls several functions in the background to create nodes, objects, markers and loads in Exudyn.
 22#all quantities are giving as 3D lists [x,y,z] for positions, velocities, ....
 23#rotations are usually given as rotation matrix (numpy array);
 24#RotationVector2RotationMatrix([rotX, rotY, rotZ]) computes a rotation around the global x,y,z rotation axis
 25
 26#%%++++++++++++++++++++++++++++++++++++++++++++++++++++
 27#create rigid bodies and mass points with distance constraint and joints
 28import exudyn as exu
 29from exudyn.utilities import * #includes itemInterface, graphicsDataUtilities and rigidBodyUtilities
 30import numpy as np
 31SC = exu.SystemContainer()
 32mbs = SC.AddSystem() #create a MainSystem 'mbs' to work with
 33
 34#draw orthonormal cube in local frame where it is added to;
 35#cube is added to reference point of object, usually the center of mass (COM):
 36graphicsCube = GraphicsDataOrthoCubePoint(centerPoint = [0,0,0],
 37                                          size=[1,0.1,0.1], color=color4orange)
 38
 39#create inertia (mass, COM, inertia tensor) to be used in rigid body:
 40inertiaCube = InertiaCuboid(density=5000, sideLengths=[1,0.1,0.1])
 41
 42#create simple rigid body
 43#note that graphics is always attached to reference point of body, which is by default the COM
 44b0 = mbs.CreateRigidBody(inertia = inertiaCube,
 45                         referencePosition = [0.5,0,0], #reference position x/y/z of COM
 46                         referenceRotationMatrix=RotationVector2RotationMatrix([0,0,pi*0.5]),
 47                         initialAngularVelocity=[2,0,0],
 48                         initialVelocity=[0,4,0],
 49                         gravity = [0,-9.81,0],
 50                         graphicsDataList = [graphicsCube])
 51
 52#add an additional load to rigid body at left end; this always requires markers;
 53#for torques use Torque(...) instead of Force(...)
 54markerBody0 = mbs.AddMarker(MarkerBodyRigid(bodyNumber=b0, localPosition=[-0.5,0,0]))
 55mbs.AddLoad(Force(markerNumber=markerBody0,
 56                  loadVector=[10,0,0])) #load is 10N in x-direction
 57
 58#create a simple mass point at [1,-1,0] with initial velocity
 59m1 = mbs.CreateMassPoint(referencePosition=[1,-1,0],
 60                         initialVelocity = [2,5,0], #initial velocities for mass point
 61                         physicsMass=1, drawSize = 0.2)
 62#we can obtain the node number from the mass point:
 63n1 = mbs.GetObject(m1)['nodeNumber']
 64
 65#add a ground object:
 66#graphics data for sphere:
 67gGround0 = GraphicsDataSphere(point=[3,1,0], radius = 0.1, color=color4red, nTiles=16)
 68#graphics for checkerboard background:
 69gGround1 = GraphicsDataCheckerBoard(point=[3,0,-2], normal=[0,0,1], size=10)
 70oGround = mbs.AddObject(ObjectGround(visualization=VObjectGround(graphicsData=[gGround0,gGround1])))
 71
 72#create a rigid distance between bodies (using local position) or between nodes
 73mbs.CreateDistanceConstraint(bodyOrNodeList=[oGround, b0],
 74                             localPosition0 = [ 0. ,0,0],
 75                             localPosition1 = [-0.5,0,0],
 76                             distance=None, #automatically computed
 77                             drawSize=0.06)
 78
 79#distance constraint between body b0 and mass m1
 80mbs.CreateDistanceConstraint(bodyOrNodeList=[b0, m1],
 81                             localPosition0 = [0.5,0,0],
 82                             localPosition1 = [0.,0.,0.], #must be [0,0,0] for Node
 83                             distance=None, #automatically computed
 84                             drawSize=0.06)
 85
 86#add further rigid body, which will be connected with joints
 87b1 = mbs.CreateRigidBody(inertia = InertiaCuboid(density=5000, sideLengths=[1,0.1,0.1]),
 88                          referencePosition = [2.5,0,0], #reference position x/y/z
 89                          gravity = [0,-9.81,0],
 90                          graphicsDataList = [graphicsCube])
 91
 92b2 = mbs.CreateRigidBody(inertia = InertiaCuboid(density=5000, sideLengths=[1,0.1,0.1]),
 93                          referencePosition = [3.5,0,0], #reference position x/y/z
 94                          gravity = [0,-9.81,0],
 95                          graphicsDataList = [graphicsCube])
 96
 97#create revolute joint with following args:
 98    # name: name string for joint; markers get Marker0:name and Marker1:name
 99    # bodyNumbers: a list of object numbers for body0 and body1; must be rigid body or ground object
100    # position: a 3D vector as list or np.array: if useGlobalFrame=True it describes the global position of the joint in reference configuration; else: local position in body0
101    # axis: a 3D vector as list or np.array: if useGlobalFrame=True it describes the global rotation axis of the joint in reference configuration; else: local axis in body0
102    # useGlobalFrame: if False, the point and axis vectors are defined in the local coordinate system of body0
103    # show: if True, connector visualization is drawn
104    # axisRadius: radius of axis for connector graphical representation
105    # axisLength: length of axis for connector graphical representation
106    # color: color of connector
107#returns list [oJoint, mBody0, mBody1], containing the joint object number, and the two rigid body markers on body0/1 for the joint
108mbs.CreateRevoluteJoint(bodyNumbers=[b1, b2], position=[3,0,0], axis=[0,0,1], #rotation along global z-axis
109                        useGlobalFrame=True, axisRadius=0.02, axisLength=0.14)
110
111
112#create prismatic joint with following args:
113    # name: name string for joint; markers get Marker0:name and Marker1:name
114    # bodyNumbers: a list of object numbers for body0 and body1; must be rigid body or ground object
115    # position: a 3D vector as list or np.array: if useGlobalFrame=True it describes the global position of the joint in reference configuration; else: local position in body0
116    # axis: a 3D vector as list or np.array containing the global translation axis of the joint in reference configuration
117    # useGlobalFrame: if False, the point and axis vectors are defined in the local coordinate system of body0
118    # show: if True, connector visualization is drawn
119    # axisRadius: radius of axis for connector graphical representation
120    # axisLength: length of axis for connector graphical representation
121    # color: color of connector
122#returns list [oJoint, mBody0, mBody1], containing the joint object number, and the two rigid body markers on body0/1 for the joint
123mbs.CreatePrismaticJoint(bodyNumbers=[oGround, b1], position=[2,0,0], axis=[1,0,0], #can move in global x-direction
124                         useGlobalFrame=True, axisRadius=0.02, axisLength=1)
125
126# #instead of the prismatic joint, we could add another revolute joint to b1 to get a double-pendulum:
127# mbs.CreateRevoluteJoint(bodyNumbers=[oGround, b1], position=[2,0,0], axis=[0,0,1],
128#                         useGlobalFrame=True, axisRadius=0.02, axisLength=0.14)
129
130
131#create simple mass point, connected with ground
132m2 = mbs.CreateMassPoint(referencePosition = [7,2,0],
133                         physicsMass = 10, gravity = [0,-9.81,0],
134                         drawSize = 0.5, color=color4blue)
135
136#create spring damper between bodies (using local position) or between nodes
137#spring-damper may not have size 0; spring reference length is computed from reference configuration
138oSD = mbs.CreateSpringDamper(bodyOrNodeList=[oGround, m2],
139                             localPosition0=[6,0,0],
140                             localPosition1=[0,0,0],
141                             stiffness=1e3, damping=1e1,
142                             drawSize=0.2)
143
144#alternatively, we can use a CartesianSpringDamper; has spring and damper coefficients as list of x/y/z components
145#it has no reference length and acts on the coordinates of both objects:
146oCSD = mbs.CreateCartesianSpringDamper(bodyOrNodeList=[oGround, m2],
147                              localPosition0=[7,2,0],
148                              localPosition1=[0,0,0],
149                              stiffness=[20,0,1e4], #stiffness in x/y/z direction
150                              damping=[0.1,0,10],
151                              drawSize=0.2)
152
153#prepare mbs for simulation:
154mbs.Assemble()
155#some simulation parameters:
156simulationSettings = exu.SimulationSettings() #takes currently set values or default values
157simulationSettings.timeIntegration.numberOfSteps = 1000
158simulationSettings.timeIntegration.endTime = 5
159
160#for redundant constraints, the following two settings:
161simulationSettings.linearSolverSettings.ignoreSingularJacobian=True
162simulationSettings.linearSolverType = exu.LinearSolverType.EigenDense
163
164mbs.SolveDynamic(simulationSettings = simulationSettings,
165                 solverType=exu.DynamicSolverType.GeneralizedAlpha)
166SC.visualizationSettings.nodes.drawNodesAsPoint=False #draw nodes as spheres; better graphics for nodes
167
168#visualize results:
169mbs.SolutionViewer()
