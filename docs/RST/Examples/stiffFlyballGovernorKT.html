  2# This is an EXUDYN example
  3#
  4# Details:  Stiff flyball governor (iftomm benchmark problem) with kinematic tree
  5#           Ref.: https://www.iftomm-multibody.org/benchmark/problem/Stiff_flyball_governor/

  7# Author:   Johannes Gerstmayr

 14import sys
 15sys.exudynFast=True
 16
 17import exudyn as exu
 18from exudyn.itemInterface import *
 19from exudyn.utilities import *
 20from exudyn.graphicsDataUtilities import *

 22#from modelUnitTests import ExudynTestStructure, exudynTestGlobals
 23import numpy as np
 24from numpy import linalg as LA

 26SC = exu.SystemContainer()
 27mbs = SC.AddSystem()
 28
 29useGraphics=False
 30
 31color = [0.1,0.1,0.8,1]
 32r = 0.2 #radius
 33L = 1   #length
 34
 35
 36#%%%%%%%%%
 37
 38background0 = GraphicsDataRectangle(-L,-L,L,L,color)
 39oGround=mbs.AddObject(ObjectGround(referencePosition= [0,0,0], visualization=VObjectGround(graphicsData= [background0])))
 40
 41
 42###############################################################################
 43## body dimensions according to reference in m
 44
 45# shaft
 46lengthShaft = 1     #z
 47widthShaft  = 0.01  #=height
 48
 49# rod
 50lengthRod = 1
 51widthRod  = 0.01    #=height
 52
 53# slider
 54dimSlider = 0.1 #x=y=z
 55sSlider = 0.5
 56
 57# scalar distance between point A and B
 58xAB = 0.1
 59beta0 = np.deg2rad(30)
 60initAngleRod = np.deg2rad(60)
 61
 62# initial angular velocity of shaft and slider
 63omega0 = [0., 0., 2*np.pi]
 64
 65
 66
 67###############################################################################
 68## body masses according to reference in kg
 69
 70density = 3000
 71
 72mShaft        = 0.3
 73mRod          = 0.3
 74mSlider       = 3
 75mMassPoint    = 5
 76mRodMassPoint = mRod + mMassPoint
 77
 78
 79###############################################################################
 80# gravity
 81g = [0,0,-9.81]
 82
 83#setup rod along x-direction
 84iRod = InertiaCuboid(density=density, sideLengths=[lengthRod,widthRod,0.01]).Translated([lengthRod/2,0,0])
 85iMass = InertiaMassPoint(mass=mMassPoint).Translated([lengthRod,0,0])
 86iRodSum = iRod+iMass
 87
 88# #compute reference point of rod (midpoint)
 89# refRod = -iRodSum.com
 90# iRodSum = iRodSum.Translated(refRod)
 91
 92# exu.Print("iRodSum=", iRodSum)
 93
 94
 95nRigidBodyNodes = 4
 96
 97#w.r.t. center of mass:
 98inertiaList=[InertiaCuboid(density=density, sideLengths=[widthShaft,widthShaft,lengthShaft]),
 99             InertiaCuboid(density=density, sideLengths=[dimSlider,dimSlider,dimSlider]),
100             iRodSum, iRodSum]
101
102#GraphicsDataOrthoCube(xMin, yMin, zMin, xMax, yMax, zMax, color=[0.,0.,0.,1.]):
103#graphicsRod    = GraphicsDataOrthoCube(-lengthRod/2,-widthRod/2,-widthRod/2, lengthRod/2,widthRod/2,widthRod/2, [0.1,0.1,0.8,1])
104graphicsShaft  = GraphicsDataOrthoCube(-widthShaft/2,-widthShaft/2,-lengthShaft/2, widthShaft/2,widthShaft/2,lengthShaft/2, [0.1,0.1,0.8,1])
105graphicsSlider = GraphicsDataOrthoCube(-dimSlider/2,-dimSlider/2,-dimSlider/2, dimSlider/2,dimSlider/2,dimSlider/2, [0.1,0.1,0.8,1])
106graphicsRodAC  = GraphicsDataOrthoCubePoint([0.5*lengthRod, 0, 0], [lengthRod,widthRod,widthRod], color4red)
107graphicsRodBD  = GraphicsDataOrthoCubePoint([0.5*lengthRod, 0, 0], [lengthRod,widthRod,widthRod], color4dodgerblue)
108
109#lists for 4 nodes/bodies: [shaft, slider, rodAC, rodBD]
110graphicsList=[[graphicsShaft], [graphicsSlider], [graphicsRodAC], [graphicsRodBD]]
111
112
113
114#create node for unknowns of KinematicTree
115nGeneric = mbs.AddNode(NodeGenericODE2(referenceCoordinates=[0.]*nRigidBodyNodes,
116                                       initialCoordinates=[0.]*nRigidBodyNodes,
117                                       initialCoordinates_t=[omega0[2],0,0,0], #initial angular velocity
118                                       numberOfODE2Coordinates=nRigidBodyNodes))
119
120#create KinematicTree
121refPosList=[[0,0,lengthShaft*0.5],        # shaft
122            [0,0,sSlider-lengthShaft*0.5],              # slider
123            [ xAB/2, 0, lengthShaft*0.5],   # rodAC
124            [-xAB/2, 0, lengthShaft*0.5]]   # rodBD
125
126
127jointTypes = [exu.JointType.RevoluteZ, exu.JointType.PrismaticZ, exu.JointType.RevoluteY, exu.JointType.RevoluteY]
128linkMasses = []
129linkCOMs = exu.Vector3DList()
130linkInertiasCOM=exu.Matrix3DList()
131
132jointTransformations=exu.Matrix3DList()
133jointOffsets = exu.Vector3DList()
134
135for i in range(nRigidBodyNodes):
136    inertia = inertiaList[i]
137    linkMasses += [inertia.Mass()]
138    linkCOMs.Append(inertia.COM())
139    linkInertiasCOM.Append(inertia.InertiaCOM())
140
141    A = np.eye(3)
142    if i == 2:
143        A = RotationMatrixY(beta0)
144    if i == 3:
145        A = RotationMatrixY((pi-beta0))
146
147
148    jointTransformations.Append(A)
149    jointOffsets.Append(refPosList[i])
150
151
152
153
154oKT=mbs.AddObject(ObjectKinematicTree(nodeNumber=nGeneric, jointTypes=jointTypes,
155                                  linkParents=[-1,0,0,0],
156                                  jointTransformations=jointTransformations, jointOffsets=jointOffsets,
157                                  linkInertiasCOM=linkInertiasCOM, linkCOMs=linkCOMs, linkMasses=linkMasses,
158                                  baseOffset = [0.,0.,0.], gravity=g,
159                                  visualization=VObjectKinematicTree(graphicsDataList = graphicsList)
160                                  ))
161
162
163
164
165###############################################################################
166## spring-damper parameters for connecting the rods with the slider
167
168# spring
169k  = 8.e5 # spring stiffness in N/m
170l0 = 0.5  # relaxed spring length in m
171
172# damper
173c = 4.e4
174
175markerRodACSlider = mbs.AddMarker(MarkerKinematicTreeRigid(objectNumber=oKT, linkNumber=2,
176                                                           localPosition=[lengthRod/2,0,0]))
177markerSliderPointE = mbs.AddMarker(MarkerKinematicTreeRigid(objectNumber=oKT, linkNumber=1,
178                                                           localPosition=[dimSlider/2,0,0]))
179
180markerRodBDSlider = mbs.AddMarker(MarkerKinematicTreeRigid(objectNumber=oKT, linkNumber=3,
181                                                           localPosition=[lengthRod/2,0,0]))
182markerSliderPointF = mbs.AddMarker(MarkerKinematicTreeRigid(objectNumber=oKT, linkNumber=1,
183                                                           localPosition=[-dimSlider/2,0,0]))
184
185mbs.AddObject(SpringDamper(markerNumbers=[markerSliderPointE, markerRodACSlider], stiffness=k, damping=c, referenceLength=l0))
186mbs.AddObject(SpringDamper(markerNumbers=[markerSliderPointF, markerRodBDSlider], stiffness=k, damping=c, referenceLength=l0))
187
188
189sPos = mbs.AddSensor(SensorKinematicTree(objectNumber=oKT, linkNumber=1,
190                                         localPosition=[0,0,0],storeInternal=True,
191                                         outputVariableType=exu.OutputVariableType.Position))
192
193
