  4# Details:  Double pendulum with added torques for control;

 12#+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
 13
 14
 15import exudyn as exu
 16from exudyn.utilities import *
 17from math import sin, cos, pi
 18
 19SC = exu.SystemContainer()
 20mbs = SC.AddSystem()
 21
 22# create nodes:
 23n0=mbs.AddNode(NodeRigidBody2D(referenceCoordinates=[0,0,0], initialVelocities=[0,0,0]))
 24n1=mbs.AddNode(NodeRigidBody2D(referenceCoordinates=[1,0,0], initialVelocities=[0,0,0]))
 25
 26# create bodies:
 27b0=mbs.AddObject(RigidBody2D(physicsMass=1, physicsInertia=1,nodeNumber=n0,
 28       visualization=VRigidBody2D(graphicsData=[GraphicsDataLine([[-0.5,0,0],[0.5,0,0]])])))
 29b1=mbs.AddObject(RigidBody2D(physicsMass=1, physicsInertia=1,nodeNumber=n1,
 30                 visualization=VRigidBody2D(graphicsData=[GraphicsDataLine([[-0.5,0,0],[0.5,0,0]])])))
 31
 32# add markers and loads:
 33oGround = mbs.AddObject(ObjectGround())
 34mGround=mbs.AddMarker(MarkerBodyRigid(bodyNumber=oGround, localPosition=[-0.5, 0., 0.]))
 35m00 = mbs.AddMarker(MarkerBodyRigid(bodyNumber=b0, localPosition=[-0.5, 0., 0.]))
 36m01 = mbs.AddMarker(MarkerBodyRigid(bodyNumber=b0, localPosition=[ 0.5, 0., 0.]))
 37m10 = mbs.AddMarker(MarkerBodyRigid(bodyNumber=b1, localPosition=[-0.5, 0., 0.]))
 38#m12 = mbs.AddMarker(MarkerBodyRigid(bodyNumber=b1, localPosition=[ 0.5, 0., 0.]))
 39
 40# add joints:
 41jg0 = mbs.AddObject(RevoluteJoint2D(markerNumbers=[mGround,m00]))
 42j01 = mbs.AddObject(RevoluteJoint2D(markerNumbers=[m01,m10]))
 43
 44# add loads:
 45
 46mLoad0=mbs.AddMarker(MarkerNodePosition(nodeNumber=n0))
 47mLoad1=mbs.AddMarker(MarkerNodePosition(nodeNumber=n1))
 48
 49mbs.AddLoad(Force(markerNumber=mLoad0, loadVector=[0,-9.81*1,0]))
 50mbs.AddLoad(Force(markerNumber=mLoad1, loadVector=[0,-9.81*1,0]))
 51
 52#%%++++++++++++++++++++++++++++++++++++++++++
 53#add controller using loads
 54nGround = mbs.AddNode(NodePointGround(visualization=VNodePointGround(show=False)) )
 55mCGround= mbs.AddMarker(MarkerNodeCoordinate(nodeNumber=nGround, coordinate=0))
 56mRotNode0 = mbs.AddMarker(MarkerNodeCoordinate(nodeNumber=n0, coordinate=2))
 57mRotNode1 = mbs.AddMarker(MarkerNodeCoordinate(nodeNumber=n1, coordinate=2))
 58
 59def UFtorque1(mbs, t, load):
 60    omega = 0.5
 61    phi0 = mbs.GetMarkerOutput(mRotNode0, exu.OutputVariableType.Coordinates)[0]
 62    deltaPhi = mbs.GetMarkerOutput(mRotNode1, exu.OutputVariableType.Coordinates)[0] - phi0
 63
 64    phi0_t = mbs.GetMarkerOutput(mRotNode0, exu.OutputVariableType.Coordinates_t)[0]
 65    deltaPhi_t = mbs.GetMarkerOutput(mRotNode1, exu.OutputVariableType.Coordinates_t)[0] - phi0_t
 66
 67    phiDesired = -pi*0.5*sin(omega*pi*t)
 68    phiDesired_t = -pi*0.5*omega*cos(omega*pi*t)
 69
 70    k = 5000
 71    d = 0.02*k
 72    torque = k*(phiDesired-deltaPhi) + d*(phiDesired_t-deltaPhi_t)
 73    return torque
 74
 75def UFtorque0(mbs, t, load):
 76    phi0 = mbs.GetMarkerOutput(mRotNode0, exu.OutputVariableType.Coordinates)[0]
 77    omega = 0.5
 78    phiDesired = pi*0.5*sin(omega*pi*t)
 79    phi0_t = mbs.GetMarkerOutput(mRotNode0, exu.OutputVariableType.Coordinates_t)[0]
 80    phiDesired_t = pi*0.5*omega*cos(omega*pi*t)
 81    k = 5000 #*100
 82    d = 0.02*k
 83    torque = k*(phiDesired-phi0) + d*(phiDesired_t-phi0_t)
 84
 85    return torque - UFtorque1(mbs,t,0) #torque1 also acting on body0
 86
 87mbs.AddLoad(LoadCoordinate(markerNumber = mRotNode0,
 88                           load = 1, loadUserFunction = UFtorque0))
 89
 90
 91mbs.AddLoad(LoadCoordinate(markerNumber = mRotNode1,
 92                           load = 1, loadUserFunction = UFtorque1))
 93
 94
 95# add time integration scheme:
 96mbs.Assemble()
 97simulationSettings = exu.SimulationSettings()
 98simulationSettings.timeIntegration.numberOfSteps = 1000
 99simulationSettings.timeIntegration.endTime = 5
100simulationSettings.timeIntegration.simulateInRealtime =q True
101simulationSettings.solutionSettings.writeSolutionToFile = True
102simulationSettings.solutionSettings.solutionWritePeriod = 0.01
103
104simulationSettings.timeIntegration.verboseMode = 1
105simulationSettings.displayStatistics = True
106simulationSettings.displayComputationTime = True
107
108exu.StartRenderer()
109mbs.WaitForUserToContinue()
110exu.SolveDynamic(mbs,simulationSettings)
111exu.StopRenderer()
