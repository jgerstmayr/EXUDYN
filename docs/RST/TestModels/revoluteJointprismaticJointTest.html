 13import exudyn as exu
 14from exudyn.utilities import * #includes itemInterface and rigidBodyUtilities
 15import exudyn.graphics as graphics #only import if it does not conflict
 16
 17from math import sin, cos, pi
 18import numpy as np
 19
 20useGraphics = True #without test
 21#+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
 22#you can erase the following lines and all exudynTestGlobals related operations if this is not intended to be used as TestModel:
 23try: #only if called from test suite
 24    from modelUnitTests import exudynTestGlobals #for globally storing test results

 27    class ExudynTestGlobals:

 32SC = exu.SystemContainer()
 33mbs = SC.AddSystem()
 34
 35
 36#background
 37color = [0.1,0.1,0.8,1]
 38L = 0.4 #length of bodies
 39d = 0.1 #diameter of bodies
 40
 41#create background, in order to have according zoom all
 42zz=2*L
 43#background0 = GraphicsDataRectangle(-zz,-zz,zz,zz,graphics.color.white)
 44oGround=mbs.AddObject(ObjectGround(referencePosition= [0,0,0])) #,visualization=VObjectGround(graphicsData= [background0])))
 45mPosLast = mbs.AddMarker(MarkerBodyRigid(bodyNumber = oGround, localPosition=[0,0,0]))
 46A0 = RotationMatrixX(0)
 47Alast = A0 #previous marker
 48
 49A0 = RotationMatrixX(0)
 50A1 = RotationMatrixY(0.5*pi)
 51A2 = RotationMatrixZ(0.5*pi)
 52A3 = RotationMatrixX(-0.5*pi)
 53A4 = RotationMatrixZ(0.5*pi)
 54
 55Alist=[A0,A1,A2,A3,A4]
 56
 57p0 = [0.,0.,0] #reference position
 58vLoc = np.array([L,0,0]) #last to next joint
 59#g = [0,0,-9.81]
 60g = [0,-9.81,0]
 61
 62#create a chain of bodies:
 63for i in range(5):
 64    #print("Build Object", i)
 65    inertia = InertiaCuboid(density=1000, sideLengths=[L,d,d])
 66    #p0 += Alist[i] @ (0.5*vLoc)
 67    p0 += (0.5*vLoc)
 68
 69    ep0 = eulerParameters0 #no rotation
 70    graphicsBody = graphics.Brick([0,0,0], [L,d,d], graphics.color.steelblue)
 71    oRB = mbs.CreateRigidBody(inertia=inertia,
 72                              referencePosition=p0,
 73                              referenceRotationMatrix=A0,
 74                              gravity=g,
 75                              graphicsDataList=[graphicsBody])
 76    nRB= mbs.GetObject(oRB)['nodeNumber']
 77
 78    mPos0 = mbs.AddMarker(MarkerBodyRigid(bodyNumber = oRB, localPosition = [-0.5*L,0,0]))
 79    mPos1 = mbs.AddMarker(MarkerBodyRigid(bodyNumber = oRB, localPosition = [ 0.5*L,0,0]))
 80    useGenericJoint = False #for comparison
 81    if True:
 82        if i != 2:
 83            if not useGenericJoint:
 84                mbs.AddObject(ObjectJointRevoluteZ(markerNumbers = [mPosLast, mPos0],
 85                                                  rotationMarker0=Alist[i],
 86                                                  rotationMarker1=Alist[i],
 87                                                  visualization=VObjectJointRevoluteZ(axisRadius=0.5*d, axisLength=1.2*d)
 88                                                  ))
 89            else: #compare to GenericJoint
 90                mbs.AddObject(ObjectJointGeneric(markerNumbers = [mPosLast, mPos0],
 91                                                 constrainedAxes=[1,1,1,1,1,0],
 92                                                 rotationMarker0=Alist[i],
 93                                                 rotationMarker1=Alist[i],
 94                                                 visualization=VGenericJoint(axesRadius=0.5*d, axesLength=1.2*d)
 95                                                 ))
 96        else:
 97            if not useGenericJoint:
 98                mbs.AddObject(ObjectJointPrismaticX(markerNumbers = [mPosLast, mPos0],
 99                                                  rotationMarker0=Alist[i],
100                                                  rotationMarker1=Alist[i],
101                                                  visualization=VObjectJointPrismaticX(axisRadius=0.5*d, axisLength=1.2*d)
102                                                  ))
103            else: #compare to GenericJoint
104                mbs.AddObject(ObjectJointGeneric(markerNumbers = [mPosLast, mPos0],
105                                                 constrainedAxes=[0,1,1,1,1,1],
106                                                 rotationMarker0=Alist[i],
107                                                 rotationMarker1=Alist[i],
108                                                 visualization=VGenericJoint(axesRadius=0.5*d, axesLength=1.2*d)
109                                                 ))
110            #add spring, to limit motion in prismatic joint:
111            k=500
112            damp=k*0.02
113            mbs.AddObject(CartesianSpringDamper(markerNumbers = [mPosLast, mPos0],
114                                       stiffness = [k]*3, damping = [damp]*3,
115                                       visualization=VCartesianSpringDamper(show=False, drawSize=0.5*d)))
116
117    mPosLast = mPos1
118
119    # p0 += Alist[i] @ (0.5*vLoc)
120    p0 += (0.5*vLoc)
121    Alast = Alist[i]
122
123mbs.AddLoad(LoadForceVector(markerNumber=mPosLast, loadVector=[0,0,20]))
124
125mbs.Assemble()

145SC.visualizationSettings.nodes.show = True
146SC.visualizationSettings.nodes.drawNodesAsPoint  = False
147SC.visualizationSettings.nodes.showBasis = True
148SC.visualizationSettings.nodes.basisSize = 0.015
